// ====================== TEAM MEMBERS + 3D ====================== function loadTeamMembers(){ const tM = JSON.parse(localStorage.getItem("teamMembers") || "[]"); const grid = document.getElementById("teamGrid"); const manager = document.getElementById("teamManager"); if(grid) grid.innerHTML = ""; if(manager) manager.innerHTML = ""; tM.forEach((m,i)=>{ // ===== FRONT-END CARD ===== const div = document.createElement("div"); div.className = "member"; div.setAttribute("data-contact", `${m.email || 'No email'}, ${m.phone || 'No phone'}` ); // 3D Container const logoContainer = document.createElement("div"); logoContainer.className = "main-logo-container"; // Base Layer (shadow layer) const base = document.createElement("img"); base.src = m.img || "maroon.jpg"; base.style.transform = "translateZ(0px) scale(1.05)"; base.style.filter = "brightness(.7)"; base.style.opacity = ".4"; // Middle Layer const mid = document.createElement("img"); mid.src = m.img || "maroon.jpg"; mid.style.transform = "translateZ(15px)"; mid.style.opacity = ".7"; // Top Layer const top = document.createElement("img"); top.src = m.img || "maroon.jpg"; top.style.transform = "translateZ(35px)"; logoContainer.appendChild(base); logoContainer.appendChild(mid); logoContainer.appendChild(top); div.appendChild(logoContainer); const h = document.createElement("h3"); h.textContent = m.name || "No Name"; div.appendChild(h); const p = document.createElement("p"); p.textContent = m.role || "No Role"; div.appendChild(p); if(grid) grid.appendChild(div); // ===== ADMIN CARD ===== if(manager && document.getElementById("teamTab").classList.contains("active")){ const card = document.createElement("div"); card.className = "admin-card"; card.innerHTML = ` ${m.name}
${m.role}
Email: ${m.email || ''}
Phone: ${m.phone || ''}
`; manager.appendChild(card); } }); // ===== SAFE VANILLA TILT INIT ===== if(typeof VanillaTilt !== "undefined"){ document.querySelectorAll(".main-logo-container").forEach(el=>{ if(el.vanillaTilt) el.vanillaTilt.destroy(); }); VanillaTilt.init( document.querySelectorAll(".main-logo-container"), { max:20, speed:400, glare:true, "max-glare":0.3, perspective:1000 } ); } } // ===== DELETE TEAM MEMBER (YOU WERE MISSING THIS) ===== function deleteTeamMember(i){ const tM = JSON.parse(localStorage.getItem("teamMembers") || "[]"); tM.splice(i,1); localStorage.setItem("teamMembers", JSON.stringify(tM)); loadTeamMembers(); }
